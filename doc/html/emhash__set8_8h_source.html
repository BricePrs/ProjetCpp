424  size_type get_diss(size_type bucket, size_type next_bucket, const size_type slots) const
425  {
426  auto pbucket = reinterpret_cast<uint64_t>(&_pairs[bucket]);
427  auto pnext = reinterpret_cast<uint64_t>(&_pairs[next_bucket]);
428  if (pbucket / EMH_CACHE_LINE_SIZE == pnext / EMH_CACHE_LINE_SIZE)
429  return 0;
430  size_type diff = pbucket > pnext ? (pbucket - pnext) : (pnext - pbucket);
431  if (diff / EMH_CACHE_LINE_SIZE < slots - 1)
432  return diff / EMH_CACHE_LINE_SIZE + 1;
433  return slots - 1;
434  }

437  {

443  if (next_bucket == main_bucket)

454  break;

457  ibucket_size ++;
458  next_bucket = nbucket;
459  }
460  return (int)ibucket_size;
